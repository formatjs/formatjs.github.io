Intl.PluralRules

A spec-compliant polyfill for Intl.PluralRules fully tested by the official ECMAScript Conformance test suite

Installation#

  • npm
  • yarn
npm i @formatjs/intl-pluralrules

Requirements#

If you're supporting IE11-, this requires Intl.getCanonicalLocales.

Usage#

Simple#

import '@formatjs/intl-pluralrules/polyfill'
import '@formatjs/intl-pluralrules/locale-data/en' // locale-data for en

Dynamic import + capability detection#

import {shouldPolyfill} from '@formatjs/intl-pluralrules/should-polyfill'
async function polyfill(locale: string) {
if (shouldPolyfill()) {
// Load the polyfill 1st BEFORE loading data
await import('@formatjs/intl-pluralrules/polyfill')
}
if (Intl.PluralRules.polyfilled) {
switch (locale) {
default:
await import('@formatjs/intl-pluralrules/locale-data/en')
break
case 'fr':
await import('@formatjs/intl-pluralrules/locale-data/fr')
break
}
}
}
